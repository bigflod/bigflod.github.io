Ejemplo 4


-- Esto es un comentario en VHDL
library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity AND_Gate is
    Port ( A : in  STD_LOGIC;
           B : in  STD_LOGIC;
           Y : out STD_LOGIC);
end AND_Gate;

architecture Behavioral of AND_Gate is
begin
    Y <= A and B;
end Behavioral;

Ejemplo 5


#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    std::vector<int> numeros = {5, 2, 9, 1, 7};

    std::cout << "Números originales: ";
    for(int n : numeros) {
        std::cout << n << " ";
    }
    std::cout << std::endl;

    std::sort(numeros.begin(), numeros.end());

    std::cout << "Números ordenados: ";
    for(int n : numeros) {
        std::cout << n << " ";
    }
    std::cout << std::endl;

    return 0;
}
